SELECT
{
	[Measures].[Quantity Invoiced]
} ON COLUMNS,
NON EMPTY
{
	distinct(
		[Distributor - GS Only].[Distributor Name].[All].Children,
		[Distributor - GS Only].[Distributor Number].[All].Children
	)
}
ON ROWS
FROM SalesOrders
WHERE (
	STRTOMEMBER(@SalesGeoMemberKey)//[Sales Geography].[Sales Geography].[Area].&[C561]
)
